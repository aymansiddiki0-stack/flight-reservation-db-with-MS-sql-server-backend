CREATE VIEW dbo.[Flight_seats_reports]
AS 
	SELECT 
		[AIRLINE].[CompanyName], 
		[FLIGHTS].[FlightID], 
		[FLIGHTS].[DepAirportCode], 
		[FLIGHTS].[DepartureDate], 
		[FLIGHTS].[ArrivalAirportCode], 
		[FLIGHTS].[ArrivalDate], 
		[FLIGHTSEAT].[Class], 
		[FLIGHTSEAT].[SeatNumber], 
		[FLIGHTSEAT].[SeatPrice]
	FROM ([AIRLINE] 
		INNER JOIN [FLIGHTS] 
		ON [AIRLINE].[CompanyID] = [FLIGHTS].[CompanyID]) 
		INNER JOIN [FLIGHTSEAT] 
		ON [FLIGHTS].[FlightID] = [FLIGHTSEAT].[FlightID]
	GROUP BY 
		[AIRLINE].[CompanyName], 
		[FLIGHTS].[FlightID], 
		[FLIGHTS].[DepAirportCode], 
		[FLIGHTS].[DepartureDate], 
		[FLIGHTS].[ArrivalAirportCode], 
		[FLIGHTS].[ArrivalDate], 
		[FLIGHTSEAT].[Class], 
		[FLIGHTSEAT].[SeatNumber], 
		[FLIGHTSEAT].[SeatPrice]
